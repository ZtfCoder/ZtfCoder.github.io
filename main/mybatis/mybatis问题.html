动态模糊搜索 #

第一种 利用mysql 函数拼接

xml
<if test="title !=null and title!='' ">
    s.title_ like CONCAT('%',CONCAT(#{title},'%'))
</if>

xml
<if test="title !=null and title!='' ">
    <bind name="title" value="'%' + title + '%'"/>
    s.title_ like #{title}
</if>
